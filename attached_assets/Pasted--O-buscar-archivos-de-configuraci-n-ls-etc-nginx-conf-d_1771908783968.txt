server {
    listen 80;
    server_name shelties.dovexmx.com 74.208.218.225;
    
    client_max_body_size 50M;
    
    location /uploads/ {
        alias /var/www/sheltiescrm/uploads/;
        expires 30d;
        add_header Cache-Control "public, immutable";
    }
    
    location / {
        proxy_pass http://127.0.0.1:5000;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
server {
    listen 80;
    server_name *.comite.dovexmx.com comite.dovexmx.com;
    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl;
    server_name *.comite.dovexmx.com comite.dovexmx.com;
    
    ssl_certificate /etc/letsencrypt/live/comite.dovexmx.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/comite.dovexmx.com/privkey.pem;
    
    # Extraer subdominio
    set $subdomain "";
    if ($host ~* ^([^.]+)\.comite\.dovexmx\.com$) {
        set $subdomain $1;
    }
    
    location / {
        proxy_pass http://localhost:5001;
    proxy_http_version 1.1;
    proxy_set_header Upgrade $http_upgrade;
    proxy_set_header Connection 'upgrade';
    proxy_set_header Host $host;
    proxy_set_header X-Real-IP $remote_addr;
    proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    proxy_set_header X-Forwarded-Proto $scheme;
    proxy_set_header Cookie $http_cookie;
    proxy_pass_header Set-Cookie;
    proxy_cache_bypass $http_upgrade;
    }
}
